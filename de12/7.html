  7. screen = player.getscreen() #絵が描けるスクリーンを作る
  9. for i in range(3) : #4回実行する
  10.     player.forward(100) #100pxほ移動する
  11.     player.left(120) #90度に曲げる
  13. screen.listen() #プログラム活性
  14. screen.mainloop() #プログラムが動作し続ける状態を維持する
