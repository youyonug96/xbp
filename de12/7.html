1. import turtle #turtleを実行するための設定
  3. player = turtle.Turtle() #turtleを作る
  4. player.color("purple") #好きな色を決める
  5. player.shape("turtle") #模様を決める（例：turtle）
  6. player.speed(0) #実行速度（数字が少ないほど早い）
  7. screen = player.getscreen() #絵を描けるスクリーンを作る
  10. for i in range(50) : #実行回数
  11.     player.forward(100) #100px移動する
  12.     player.left(130) #130度曲がる
  14. screen.listen() #プログラム活性化
  15. screen.mainloop() #ムログラムが実行し続ける
